--categories
CREATE TABLE IF NOT EXISTS categories_history (
    ch_id          INTEGER PRIMARY KEY AUTOINCREMENT,
    category_id    INTEGER NOT NULL,
    category_brand TEXT NOT NULL,
    model_name     TEXT NOT NULL,
    product_type   TEXT NOT NULL,
    screen_size    REAL,
    color          TEXT,
    change_type    TEXT NOT NULL CHECK (change_type IN ('add','edit','delete')),
    changed_at     DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Trigger: AFTER INSERT
CREATE TRIGGER trg_categories_after_insert
AFTER INSERT ON categories
FOR EACH ROW
BEGIN
    INSERT INTO categories_history(
        category_id, category_brand, model_name, product_type, screen_size, color, change_type
    )
    VALUES (
        NEW.category_id, NEW.category_brand, NEW.model_name, NEW.product_type, NEW.screen_size, NEW.color, 'add'
    );
END;
CREATE TRIGGER trg_categories_after_update
AFTER UPDATE ON categories
FOR EACH ROW
BEGIN
    INSERT INTO categories_history(
        category_id, category_brand, model_name, product_type, screen_size, color, change_type
    )
    VALUES (
        NEW.category_id, NEW.category_brand, NEW.model_name, NEW.product_type, NEW.screen_size, NEW.color, 'edit'
    );
END;
CREATE TRIGGER trg_categories_before_delete
BEFORE DELETE ON categories
FOR EACH ROW
BEGIN
    SELECT
        CASE
            WHEN EXISTS (
                SELECT 1 FROM products WHERE category_id = OLD.category_id
            )
            THEN
                RAISE(ABORT, 'Cannot delete category; products exist under this category')
        END;
END;
CREATE TRIGGER trg_categories_after_delete
AFTER DELETE ON categories
FOR EACH ROW
BEGIN
    INSERT INTO categories_history(
        category_id, category_brand, model_name, product_type, screen_size, color, change_type
    )
    VALUES (
        OLD.category_id, OLD.category_brand, OLD.model_name, OLD.product_type, OLD.screen_size, OLD.color, 'delete'
    );
END;

--order_items
CREATE TRIGGER trg_order_items_after_insert
AFTER INSERT ON order_items
FOR EACH ROW
BEGIN
    -- Update inventory table: reduce total_qty
    UPDATE inventory
    SET total_qty = total_qty - NEW.qty,
        inventory_amount = (total_qty - NEW.qty) * unit_cost,
        last_updated = CURRENT_TIMESTAMP
    WHERE product_id = NEW.product_id;

    -- Optionally, update stock at specific location
    UPDATE stocks
    SET qty_stocked = qty_stocked - NEW.qty,
        stock_line_amount = (qty_stocked - NEW.qty) * unit_cost,
        updated_at = CURRENT_TIMESTAMP
    WHERE product_id = NEW.product_id AND location_id = (SELECT location_id FROM orders WHERE order_id = NEW.order_id);

    -- Insert into orders_history for auditing
    INSERT INTO orders_history(
        order_id,
        customer_id,
        order_date,
        order_type,
        location_id,
        status,
        subtotal,
        shipping_fee,
        total,
        payment_method,
        delivery_method,
        change_type,
        changed_by,
        changed_at,
        remarks
    )
    VALUES (
        NEW.order_id,
        (SELECT customer_id FROM orders WHERE order_id = NEW.order_id),
        (SELECT order_date FROM orders WHERE order_id = NEW.order_id),
        (SELECT order_type FROM orders WHERE order_id = NEW.order_id),
        (SELECT location_id FROM orders WHERE order_id = NEW.order_id),
        (SELECT status FROM orders WHERE order_id = NEW.order_id),
        (SELECT subtotal FROM orders WHERE order_id = NEW.order_id),
        (SELECT shipping_fee FROM orders WHERE order_id = NEW.order_id),
        (SELECT total FROM orders WHERE order_id = NEW.order_id),
        (SELECT payment_method FROM orders WHERE order_id = NEW.order_id),
        (SELECT delivery_method FROM orders WHERE order_id = NEW.order_id),
        'add',
        NULL,
        CURRENT_TIMESTAMP,
        'Order item added'
    );
END;

--returns
CREATE TRIGGER trg_returns_after_insert
AFTER INSERT ON returns
FOR EACH ROW
BEGIN
    -- Update inventory
    UPDATE inventory
    SET total_qty = total_qty + (
        SELECT SUM(qty_returned) FROM return_items WHERE return_id = NEW.return_id AND product_id = inventory.product_id
    ),
    inventory_amount = total_qty * unit_cost,
    last_updated = CURRENT_TIMESTAMP
    WHERE product_id IN (SELECT product_id FROM return_items WHERE return_id = NEW.return_id);

    -- Update stocks
    UPDATE stocks
    SET qty_stocked = qty_stocked + (
        SELECT SUM(qty_returned) FROM return_items WHERE return_id = NEW.return_id AND product_id = stocks.product_id
    ),
    stock_line_amount = qty_stocked * unit_cost,
    updated_at = CURRENT_TIMESTAMP
    WHERE location_id = NEW.location_id
      AND product_id IN (SELECT product_id FROM return_items WHERE return_id = NEW.return_id);

    -- Log negative sale in sales table
    INSERT INTO sales(
        order_id,
        order_item_id,
        return_id,
        return_item_id,
        product_id,
        quantity,
        cost_price,
        wholesale_price,
        sales_price,
        amount,
        sale_date,
        location_id,
        created_by,
        notes
    )
    SELECT
        ri.return_id, 
        ri.item_id,
        NEW.return_id,
        ri.ri_id,
        oi.product_id,
        ri.qty_returned,
        p.cost_price,
        p.wholesale_price,
        p.sale_price,
        -(ri.qty_returned * p.sale_price),
        CURRENT_TIMESTAMP,
        NEW.location_id,
        NEW.customer_id,
        'Return processed'
    FROM return_items ri
    JOIN order_items oi ON ri.item_id = oi.item_id
    JOIN products p ON ri.product_id = p.product_id
    WHERE ri.return_id = NEW.return_id;
END;

        SELECT SUM(qty_returned) FROM return_items WHERE return_id = OLD.return_id AND product_id = inventory.product_id
    ),
    inventory_amount = total_qty * unit_cost,
    last_updated = CURRENT_TIMESTAMP
    WHERE product_id IN (SELECT product_id FROM return_items WHERE return_id = OLD.return_id);

    -- Reduce stock
    UPDATE stocks
    SET qty_stocked = qty_stocked - (
        SELECT SUM(qty_returned) FROM return_items WHERE return_id = OLD.return_id AND product_id = stocks.product_id
    ),
    stock_line_amount = qty_stocked * unit_cost,
    updated_at = CURRENT_TIMESTAMP
    WHERE location_id = OLD.location_id
      AND product_id IN (SELECT product_id FROM return_items WHERE return_id = OLD.return_id);

    -- Delete negative sales related to this return
    DELETE FROM sales
    WHERE return_id = OLD.return_id;
END;

        last_updated = CURRENT_TIMESTAMP
    WHERE product_id = (SELECT product_id FROM order_items WHERE item_id = NEW.item_id);

    -- Increase stock at the location of the original order
    UPDATE stocks
    SET qty_stocked = qty_stocked + NEW.qty_returned,
        stock_line_amount = (qty_stocked + NEW.qty_returned) * unit_cost,
        updated_at = CURRENT_TIMESTAMP
    WHERE product_id = (SELECT product_id FROM order_items WHERE item_id = NEW.item_id)
      AND location_id = (SELECT location_id FROM orders WHERE order_id = (SELECT order_id FROM return_items WHERE ri_id = NEW.ri_id));

    -- Log negative sale to reflect return in sales
    INSERT INTO sales(
        return_item_id,
        return_id,
        product_id,
        quantity,
        cost_price,
        wholesale_price,
        sales_price,
        amount,
        sale_date,
        location_id,
        created_by,
        notes

        NEW.ri_id,

    JOIN returns r ON NEW.return_id = r.return_id
    JOIN orders o ON r.order_id = o.order_id

        updated_at = CURRENT_TIMESTAMP;

    INSERT INTO stock_movement_history (
        smr_id,
        product_id,
        from_location_id,
        to_location_id,
        old_quantity,
        new_quantity,
        change_type,
        changed_by,
        changed_at,
        remarks
    )
    VALUES (
        NEW.smr_id,
        NEW.product_id,

        NEW.to_location_id,
        NEW.quantity_moved,
        NEW.quantity_moved,
        'move',
        NEW.approved_by,

        'Stock moved between locations'
    );

    -- Inventory remains same total_qty, only location distribution changes, so we may skip change to total_qty
END;

AFTER UPDATE ON stock_movement_records
FOR EACH ROW

    UPDATE stocks
    SET qty_stocked = qty_stocked + OLD.quantity_moved
    WHERE product_id = OLD.product_id AND location_id = OLD.from_location_id;

    UPDATE stocks
    SET qty_stocked = qty_stocked - OLD.quantity_moved
    WHERE product_id = OLD.product_id AND location_id = OLD.to_location_id;

    -- Apply new movement
    UPDATE stocks
    SET qty_stocked = qty_stocked - NEW.quantity_moved
    WHERE product_id = NEW.product_id AND location_id = NEW.from_location_id;

    INSERT INTO stocks (product_id, location_id, qty_stocked, unit_cost, manager_id, created_at, updated_at)
    VALUES (NEW.product_id, NEW.to_location_id, NEW.quantity_moved, NEW.unit_cost, NEW.approved_by, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP)
    ON CONFLICT(product_id, location_id) DO UPDATE SET
        qty_stocked = qty_stocked + NEW.quantity_moved,
        unit_cost = NEW.unit_cost,
        updated_at = CURRENT_TIMESTAMP;

    -- Log in history
    INSERT INTO stock_movement_history (
        smr_id,
        product_id,
        from_location_id,
        to_location_id,
        old_quantity,
        new_quantity,
        change_type,
        changed_by,
        changed_at,
        remarks
    )
    VALUES (
        NEW.smr_id,
        NEW.product_id,
        NEW.from_location_id,
        NEW.to_location_id,
        OLD.quantity_moved,
        NEW.quantity_moved,
        'adjust',
        NEW.approved_by,
        CURRENT_TIMESTAMP,
        'Stock movement updated'
    );
END;

--invoices
CREATE TRIGGER trg_invoices_after_insert
AFTER INSERT ON invoices
FOR EACH ROW
BEGIN
    -- Update related order status if order_id exists
    UPDATE orders
    SET status = 'processed'
    WHERE order_id = NEW.order_id;

    -- Reduce stocks based on invoice items
    UPDATE stocks
    SET qty_stocked = qty_stocked - (
        SELECT SUM(quantity_sold) 
        FROM invoice_items 
        WHERE invoice_id = NEW.invoice_id AND product_id = stocks.product_id
    )
    WHERE product_id IN (SELECT product_id FROM invoice_items WHERE invoice_id = NEW.invoice_id);

    -- Update inventory table
    UPDATE inventory
    SET total_qty = (
        SELECT SUM(qty_stocked) 
        FROM stocks 
        WHERE product_id = inventory.product_id
    ),
    unit_cost = (
        SELECT unit_cost FROM products WHERE product_id = inventory.product_id
    ),
    inventory_amount = total_qty * unit_cost,
    last_updated = CURRENT_TIMESTAMP
    WHERE product_id IN (SELECT product_id FROM invoice_items WHERE invoice_id = NEW.invoice_id);

    -- Optional: log invoice creation in invoices_history
    INSERT INTO invoices_history (invoice_id, change_type, changed_by, changed_at, remarks)
    VALUES (NEW.invoice_id, 'add', NEW.created_by, CURRENT_TIMESTAMP, 'Invoice created');
END;
